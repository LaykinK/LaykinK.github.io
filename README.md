**FizzBuzz Challenge**
```markdown
<!DOCTYPE html>
<html>
<head>
<meta charset="UTF-8">
<title>Fizz Buzz</title>
<script>

function fizzbuzz() {
	var display = document.getElementById('display');
	var displayHTML = "";
	for (i = 1; i < 101; i++) {
		if (i % 3 == 0 && i % 5 == 0){
			displayHTML += "fizzbuzz" + "<p>";
		}
		else if (i % 3 == 0){
			displayHTML += "fizz" + "<p>";
		} 
		else if (i % 5 == 0) {
			displayHTML += "buzz" + "<p>";
		} 
		else {
			displayHTML += i + "<p>";
		}
	}
	display.innerHTML = displayHTML;
}

</script>

</head>

<body onload = "fizzbuzz()">
<div id="display">

</div>
</body>

</html>
```
